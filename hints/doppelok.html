Wrap everything in a function expression and use the max() function. 0 was set as the default argument, so that if the string is empty, our function will return 0. 
def double_substring(line: sting) -> int:
    num = 1
    result = []
    for a in range(len(line)):
        for i in range(len(line)):
            if line.count(line[i:i+num]) > 1:
                result.append(len(line[i:i+num]))
        num += 1
    return max(result, default=0)

import re

def double_substring(line: sting) -> int:
    return max(map(len, re.findall(r'(?=(.*).*\1)', line)))

def double_substring(line: sting) -> int:
    res = 0
    x = 1
    while x <= len(line):
        if line[:x] in line[x:]:
