Comment puis-je renvoyer la bonne réponse?

Enveloppez le tout dans une expression de fonction et utilisez la fonction max(). 0 a été défini comme argument par défaut, de sorte que si la chaîne est vide, notre fonction renverra 0.

def double_substring(line: str) -> int:

    num = 1
    result = []
    for a in range(len(line)):
        for i in range(len(line)):
            if line.count(line[i:i+num])> 1:
                result.append(len(line[i:i+num]))
        num += 1
    return max(result, default=0)

import re


def double_substring(line: str) -> int:

    return max(map(len, re.findall(r'(?=(.*).*\1)', line)))

def double_substring(line: str) -> int:

    res = 0
    x = 1
    while x <= len(line):
        if line[:x] in line[x:]:
            res = max(res,len(line[:x]))
            x += 1
        else:
            line = line[1:]
    return res
